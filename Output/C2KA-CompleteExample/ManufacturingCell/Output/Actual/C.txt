begin AGENT where

	Control Agent :=  LOAD + IDLE + INIT + PROC + PREP 

end


begin NEXT_BEHAVIOR  where

	(PREP,unloaded) = INIT
(IDLE,start) = LOAD
(LOAD,loaded) = PREP
(PROC,processed) = IDLE
(INIT,process) = PROC

end


begin NEXT_STIMULUS  where

	(PREP,unloaded) = setup
(IDLE,start) = load
(PROC,processed) = end
(LOAD,loaded) = prepare
(INIT,process) = done

end


begin CONCRETE_BEHAVIOR  where

	IDLE => [ state := 0 ]
	INIT => [ state := 3 ]
	PREP => [ state := 2 ]
	PROC => [ state := 4 ]
	LOAD => [ state := 1 ]
	
end
