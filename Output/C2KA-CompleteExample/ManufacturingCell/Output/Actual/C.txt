begin AGENT where

	Control Agent :=  PREP + IDLE + INIT + PROC + LOAD 

end


begin NEXT_BEHAVIOR  where

	(process,INIT) = PROC
(start,IDLE) = LOAD
(loaded,LOAD) = PREP
(unloaded,PREP) = INIT
(processed,PROC) = IDLE

end


begin NEXT_STIMULUS  where

	(loaded,LOAD) = prepare
(processed,PROC) = end
(process,INIT) = done
(unloaded,PREP) = setup
(start,IDLE) = load

end


begin CONCRETE_BEHAVIOR  where

	PREP => [ state := 2 ]
	INIT => [ state := 3 ]
	IDLE => [ state := 0 ]
	PROC => [ state := 4 ]
	LOAD => [ state := 1 ]
	
end
