begin AGENT where

	Storage Agent :=  EMPTY + FULL 

end


begin NEXT_BEHAVIOR  where

	(load,EMPTY) = FULL
(unload,FULL) = EMPTY

end


begin NEXT_STIMULUS  where

	(unload,FULL) = unloaded
(load,EMPTY) = loaded

end


begin CONCRETE_BEHAVIOR  where

	EMPTY => [ status := 0 ]
	FULL => [ status := 1 ]
	
end
