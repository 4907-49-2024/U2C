	Processing Agent :=  STBY + WORK + SET

	(SET,process) = WORK

end

	(SET,process) = processed

end

	SET => [ if (material = 1 AND state = 2 AND status = 0) -> ready:= 1 | NOT (material = 1 AND state = 2 AND status = 0) -> := 0 fi ]

WORK => [ if (ready = 1) -> part := PROCESS() | NOT (ready = 1) -> part := null fi ]
